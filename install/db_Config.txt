#DB Config File;
Client Table->CREATE TABLE clientTbl(clientId INT NOT NULL AUTO_INCREMENT, clientUsername VARCHAR(256) NOT NULL, clientPass VARCHAR(256) NOT NULL, adminHint VARCHAR(256) NULL, PRIMARY KEY ( clientId ));
Client Information->CREATE TABLE clientInfo(clientInfoId INT NOT NULL AUTO_INCREMENT, clientId INT NOT NULL, firstName VARCHAR(256) NOT NULL, lastName VARCHAR(256) NOT NULL, addRess VARCHAR(256) NULL, PRIMARY KEY ( clientInfoId ));
Document Storage->CREATE TABLE documentTbl(documentId INT NOT NULL AUTO_INCREMENT, clientId INT NOT NULL, documentCode VARCHAR(256) NOT NULL, documentKey VARCHAR(256) NULL, documentPrivacy VARCHAR(3) NULL, PRIMARY KEY ( documentId ));
Admin Table->CREATE TABLE adminTbl(adminid INT NOT NULL AUTO_INCREMENT, adminUsername VARCHAR(256) NOT NULL, adminPass VARCHAR(256) NOT NULL, adminHint VARCHAR(256) NULL, PRIMARY KEY ( adminid ));
WebsiteSettting->CREATE TABLE WebsiteSettting (settingID INT NOT NULL AUTO_INCREMENT, iconPath VARCHAR(2048) NULL, logoPath VARCHAR(2048) NULL, siteName VARCHAR(256) NOT NULL, copyRight VARCHAR(4) NOT NULL, authors VARCHAR(1024) NULL, schemeID INT(1) NOT NULL, partsID INT(1) NOT NULL, PRIMARY KEY ( settingID ) );
Page Table->CREATE TABLE PageTbl (pageID INT NOT NULL AUTO_INCREMENT, pageTitle VARCHAR(64) NOT NULL, pageDescription VARCHAR(512) NULL, pageStatus INT(1) NOT NULL, PRIMARY KEY (pageID) );
Parts Table->CREATE TABLE PartsTbl (partsID INT NOT NULL AUTO_INCREMENT, header VARCHAR(2048) NULL, footer VARCHAR(2048) NULL, sideBar VARCHAR(2048)  NULL, PRIMARY KEY (partsID) );
ColorScheme Table->CREATE TABLE ColorScheme (schemeID INT NOT NULL AUTO_INCREMENT, schemeName VARCHAR(64) NOT NULL, cssPath VARCHAR(2048) NOT NULL, description VARCHAR(512) NOT NULL, PRIMARY KEY (schemeID) );
FileDirectory Table->CREATE TABLE FileDirectory (fileID INT NOT NULL, fileName VARCHAR(64) NOT NULL, fileCode VARCHAR(6) NOT NULL, fileKey VARCHAR(4) NULL);
Transaction Table->CREATE TABLE TransactionTable (transactionID INT NOT NULL AUTO_INCREMENT, machineID INT NOT NULL, clientID INT NOT NULL, sale INT(4) NOT NULL, PRIMARY KEY (transactionID) );
Machine Table->CREATE TABLE MachineTable (machineID INT NOT NULL AUTO_INCREMENT, ownerID INT NOT NULL, machineLocation VARCHAR(512) NULL, machineStatus INT(4) NOT NULL, PRIMARY KEY (machineID) );
Owner Table->CREATE TABLE OwnerTable (ownerID INT NOT NULL AUTO_INCREMENT, ownerName VARCHAR(256) NOT NULL, address VARCHAR(512) NULL, PRIMARY KEY (ownerID) )
